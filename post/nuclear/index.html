Last updated on Aug 21, 2019 7 min read

Nessa semana a Tidy Tuesday trouxe uma base de dados de explosões nucleares e uma boa oportunidade de brincar com mapas! Como de praxe, vamos baixar e ver como estão os dados.

Aviso aos cartógrafos e profissionais de GIS: pretendo cometer várias simplificações grosseiras ao falar de projeções ao longo desse post.

Vamos começar pegando um mapa-mundi. Para nossa conveniência existe um nos pacotes ggplot e maps. Plotamos esse mapa usando a função geom_polygon() e vamos usar o tema theme_map do pacote ggthemes

world <- ggplot2::map_data("world")

world %>% 
  ggplot() +
  geom_polygon(aes(x = long, y = lat, group = group)) +
  coord_quickmap() +
  ggthemes::theme_map()

Agora vamos colocar as explosões nucleares e usar as variáveis size e color pra nos trazer mais informações. A função coord_quickmap(), como o nome indica, é boa pra visualizações rápidas. Não é exatamente equivalente a nenhuma das projeções comumente vistas em cartografia.

world %>%

Isso serve como deixa pra usar a projeção “Azimutal Equidistante” no próximo gráfico, usando a função coord_map() do ggplot2, que usa as projeções da terra definidas no pacote mapproj. Filtrei só o hemisfério norte pra esse caso. Essa projeção mantém as distâncias em relação ao pólo, mas distorce as áreas.

Agora vou deixar o mapa com uma cara um pouco mais profissional, baixando dados de melhor qualidade do pacote rworldmap, usando as funções do pacote sf, deixando o mapa com melhor qualidade. Pra mudar pra mais uma projeção, vamos usar a de Mercator dessa vez. Essa é a projeção das mais comuns, que mantém os meridianos como linhas verticais e, consequentemente, distorce todas as áreas proporcionalmente à latitude. Vou transformar nossos pontos em uma simple feature para poder converter ela na projeção de Mercator.

Daqui pra frente estou bebendo direto da fonte do Claus Wilke, seu livro Fundamentals of Data Visualization, que acabei de comprar e é espetacular, e esse gist em que ele ensina a criar mapas com diferentes projeções. Ah, o livro está inteiramente disponível online.

pacman::p_load(rgeos, rworldmap, sf) 

world_sf <- sf::st_as_sf(rworldmap::getMap(resolution = "low"))

crs_longlat <- "+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs"

crs_mercator <- "+proj=merc"

nuclear_sf <- nuclear %>% 
  drop_na(yield_lower, latitude, longitude) %>% 
  st_as_sf(coords = c("longitude", "latitude"), 
           crs = "+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs") 

nuclear_mercator <- nuclear_sf %>%
  st_transform(crs = crs_mercator)

# calculate bounding box in transformed coordinates
mercator_bbox <- 
  rbind(c(-180, -85), c(180, 85)) %>%
  st_multipoint() %>%
  st_sfc(
    crs = "+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs"
  ) %>% 
  st_transform(crs = crs_mercator)

ggplot(world_sf) + 
  geom_sf(fill = "gray95", color = "black", size = 0.5/.pt) + 
  geom_sf(data = nuclear_mercator,
          aes(size = yield_lower,
              color = country),
          show.legend = 'point',
          alpha = 0.45) +
  labs(size = "Quilotons",
       color = "País") +
  scale_x_continuous(name = NULL, breaks = seq(-120, 120, by = 60)) +
  scale_y_continuous(name = NULL, breaks = seq(-60, 60, by = 30)) +
  coord_sf(
    xlim = mercator_bbox[[1]][, 1],
    ylim = mercator_bbox[[1]][, 2],
    expand = FALSE,
    crs = crs_mercator
  ) + 
  theme_minimal() +
  theme(
    panel.background = element_rect(fill = "#56B4E950", 
                                    color = "white", size = 1),
    panel.grid.major = element_line(color = "gray30", 
                                    size = 0.25),
    legend.key.width = unit(0.05,"cm"))

Pra terminar, gostaria de testar a Projeção Descontínua de Goode, que mantém tanto áreas quanto formas das massas terrestres.

crs_goode <- "+proj=igh" # Define Projeção

# Linhas externas em latlong
lats <- c(
  90:-90, # lado direito
  -90:0, 0:-90, # terceiro corte
  -90:0, 0:-90, # segundo corte
  -90:0, 0:-90, # primeiro corte
  -90:90, # esquerda acima
  90:0, 0:90, # cortar o topo
  90 # fechar
)

longs <- c(
  rep(180, 181), # lado direito embaixo
  rep(c(80.01, 79.99), each = 91), # terceiro corte
  rep(c(-19.99, -20.01), each = 91), # segundo corte
  rep(c(-99.99, -100.01), each = 91), # primeiro corte
  rep(-180, 181), # esquerda acima
  rep(c(-40.01, -39.99), each = 91), # cortar o topo
  180 # fechar
)

goode_outline <- 
  list(cbind(longs, lats)) %>%
  st_polygon() %>%
  st_sfc(
    crs = "+proj=longlat +ellps=WGS84 +datum=WGS84 +no_defs"
  ) %>% 
  st_transform(crs = crs_goode)

# Explosões em coordenadas transformadas
nuclear_goode <- nuclear_sf %>% 
  st_transform(crs = crs_goode)

# Caixa de Limite em coordenadas transformadas
xlim <- c(-21945470, 21963330)
ylim <- c(-9538022, 9266738)

goode_bbox <- 
  list(
    cbind(
      c(xlim[1], xlim[2], xlim[2], xlim[1], xlim[1]), 
      c(ylim[1], ylim[1], ylim[2], ylim[2], ylim[1])
    )
  ) %>%
  st_polygon() %>%
  st_sfc(crs = crs_goode)

# area fora das linhas de borda
goode_without <- st_difference(goode_bbox, goode_outline)

ggplot(world_sf) + 
  geom_sf(fill = "gray95", color = "black", size = 0.5/.pt) +
  geom_sf(data = goode_without, fill = "white", color = NA) +
  geom_sf(data = goode_outline, fill = NA, color = "grey30", size = 0.5/.pt) +
  geom_sf(data = nuclear_goode,
          aes(size = yield_lower,
              color = country),
          show.legend = 'point',
          alpha = 0.45) +
  labs(size = "Quilotons",
       color = "País",
       title = "Explosões Nucleares de 1945 a 1998") +
  scale_x_continuous(name = NULL, breaks = seq(-120, 120, by = 60)) +
  scale_y_continuous(name = NULL, breaks = seq(-60, 60, by = 30)) +
  coord_sf(xlim = 0.95*xlim, ylim = 0.95*ylim, expand = FALSE, crs = crs_goode, ndiscr = 1000) + 
  theme_minimal() +
  theme(
    panel.background = element_rect(fill = "#56B4E950", color = "white", size = 1),
    panel.grid.major = element_line(color = "gray30", size = 0.25)
  )

BÔNUS: Mapa interativo

Como um bônus, sem grandes elocubrações, vou mostrar como fazer um mapa interativo usando o pacote leaflet. Fica aí só como exemplo.

pacman::p_load(leaflet)

pal <- colorFactor(
  palette = "Dark2",
  domain = world_sf$country)

world_sf %>% 
  leaflet() %>% 
  addTiles() %>% 
  addCircles(data = nuclear_sf, 
             color = ~pal(country), 
             radius = ~yield_lower*10) %>% 
  addLegend(data = nuclear_sf, 
            "bottomright", 
            pal = pal, 
            values = ~country,
            title = "País",
            opacity = 1)
